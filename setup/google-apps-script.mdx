---
title: "Deploy the Google Apps Script domain-fronting relay"
description: "Create and deploy Code.gs as a web app that fronts traffic through Google infrastructure and forwards requests to your Cloudflare Worker via UrlFetchApp."
---

The Google Apps Script relay is the domain-fronting layer. It runs on Google's infrastructure and is served under `script.google.com`, so deep packet inspection sees the TLS handshake as traffic to a well-known Google domain. The relay authenticates each request using a shared secret, then forwards it to your Cloudflare Worker exit node to reach the real target.

<Steps>
  <Step title="Open Google Apps Script">
    Go to [script.google.com](https://script.google.com) and sign in with your Google account. Click **New project**.
  </Step>

  <Step title="Clear the default code">
    Select all the default code in the editor and delete it.
  </Step>

  <Step title="Paste the relay code">
    Copy the full contents of `script/Code.gs` from the repository and paste it into the editor:

    ```javascript script/Code.gs
    /**
     * DomainFront Relay — Google Apps Script With Cloudflare Worker Exit
     *
     * FLOW:
     *   Client → GAS (Google Apps Script) → CFW (Cloudflare Worker) → Internet
     *
     * MODES:
     *   1. Single:  POST { k, m, u, h, b, ct, r }       → { s, h, b }
     *   2. Batch:   POST { k, q: [{m,u,h,b,ct,r}, ...] } → { q: [{s,h,b}, ...] }
     *
     * CHANGE THESE:
    */

    const AUTH_KEY = "STRONG_SECRET_KEY";
    const WORKER_URL = "https://example.workers.dev";

    const SKIP_HEADERS = {
      host: 1, connection: 1, "content-length": 1,
      "transfer-encoding": 1, "proxy-connection": 1, "proxy-authorization": 1,
    };

    function doPost(e) {
      try {
        var req = JSON.parse(e.postData.contents);
        if (req.k !== AUTH_KEY) return _json({ e: "unauthorized" });

        if (Array.isArray(req.q)) return _doBatch(req.q);
        return _doSingle(req);

      } catch (err) {
        return _json({ e: String(err) });
      }
    }

    function _doSingle(req) {
      if (!req.u || typeof req.u !== "string" || !req.u.match(/^https?:\/\//i)) {
        return _json({ e: "bad url" });
      }

      var payload = _buildWorkerPayload(req);

      var resp = UrlFetchApp.fetch(WORKER_URL, {
        method: "post",
        contentType: "application/json",
        payload: JSON.stringify(payload),
        muteHttpExceptions: true,
        followRedirects: true
      });

      try {
        return _json(JSON.parse(resp.getContentText()));
      } catch (e) {
        return _json({ e: "invalid worker response", raw: resp.getContentText() });
      }
    }

    function _doBatch(items) {
      var fetchArgs = [];
      var errorMap = {};

      for (var i = 0; i < items.length; i++) {
        var item = items[i];

        if (!item.u || typeof item.u !== "string" || !item.u.match(/^https?:\/\//i)) {
          errorMap[i] = "bad url";
          continue;
        }

        var payload = _buildWorkerPayload(item);

        fetchArgs.push({
          _i: i,
          _o: {
            url: WORKER_URL,
            method: "post",
            contentType: "application/json",
            payload: JSON.stringify(payload),
            muteHttpExceptions: true,
            followRedirects: true
          }
        });
      }

      var responses = [];
      if (fetchArgs.length > 0) {
        responses = UrlFetchApp.fetchAll(fetchArgs.map(function(x) { return x._o; }));
      }

      var results = [];
      var rIdx = 0;

      for (var i = 0; i < items.length; i++) {
        if (errorMap.hasOwnProperty(i)) {
          results.push({ e: errorMap[i] });
        } else {
          var resp = responses[rIdx++];
          try {
            results.push(JSON.parse(resp.getContentText()));
          } catch (e) {
            results.push({ e: "invalid worker response", raw: resp.getContentText() });
          }
        }
      }

      return _json({ q: results });
    }

    function _buildWorkerPayload(req) {
      var headers = {};

      if (req.h && typeof req.h === "object") {
        for (var k in req.h) {
          if (req.h.hasOwnProperty(k) && !SKIP_HEADERS[k.toLowerCase()]) {
            headers[k] = req.h[k];
          }
        }
      }

      return {
        u: req.u,
        m: (req.m || "GET").toUpperCase(),
        h: headers,
        b: req.b || null,
        ct: req.ct || null,
        r: req.r !== false
      };
    }

    function doGet(e) {
      return HtmlService.createHtmlOutput(
        "<!DOCTYPE html><html><head><title>My App</title></head>" +
          '<body style="font-family:sans-serif;max-width:600px;margin:40px auto">' +
          "<h1>Relay Active</h1><p>Cloudflare Worker routing enabled.</p>" +
          "</body></html>"
      );
    }

    function _json(obj) {
      return ContentService
        .createTextOutput(JSON.stringify(obj))
        .setMimeType(ContentService.MimeType.JSON);
    }
    ```
  </Step>

  <Step title="Set AUTH_KEY and WORKER_URL">
    At the top of the file, replace the two constants with your own values:

    ```javascript
    const AUTH_KEY = "your-secret-password-here";
    const WORKER_URL = "https://myworker.workers.dev";
    ```

    - `AUTH_KEY` — choose a strong, unique secret. You will enter the same value as `auth_key` in `config.json` during local proxy setup.
    - `WORKER_URL` — the full HTTPS URL of the Cloudflare Worker you deployed in the previous step.
  </Step>

  <Step title="Deploy as a web app">
    Click **Deploy** → **New deployment**. Set the deployment type to **Web app**, then configure the following:

    - **Execute as:** Me
    - **Who has access:** Anyone

    Click **Deploy** and authorize the requested permissions.
  </Step>

  <Step title="Copy the Deployment ID">
    After deploying, Google shows you a Deployment URL and a **Deployment ID** — a long alphanumeric string. Copy the Deployment ID. You will enter it as `script_id` in `config.json` when you run the local proxy setup wizard.
  </Step>
</Steps>

<Warning>
  `AUTH_KEY` in `Code.gs` must exactly match `auth_key` in your local `config.json`. Any mismatch causes every relay request to return an `"unauthorized"` error.
</Warning>

<Note>
  You can deploy the same `Code.gs` to multiple Google accounts and enter the resulting Deployment IDs as a comma-separated list in the setup wizard. The local proxy will load-balance across them using sticky-per-host selection, which spreads quota usage and improves reliability.
</Note>

<Tip>
  The free Google account quota allows approximately 20,000 `UrlFetchApp` calls per day. Each browser request may generate several relay calls, so deploying multiple scripts is recommended for heavy usage.
</Tip>
